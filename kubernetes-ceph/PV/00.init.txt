ceph osd pool create kube 32
ceph osd pool ls

rbd create kube --size 10240 --pool kube  --image-format 2 --image-feature  layering   [-m {mon-IP}] [-k /etc/ceph/ceph.client.admin.keyring]
rbd create test-pv01 --size 1024 -p kube
rbd ls --pool kube

ceph auth get-or-create client.kube mon 'allow r' osd 'allow rwx  pool=kube'

ceph auth get-key client.admin | base64
